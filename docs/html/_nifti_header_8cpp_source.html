1205  void NiftiHeader::MatToQuatern( ) {
1206  try {
1207  Matrix< double > Q( 3, 3 );
1208  COMMENT( "offset outputs are read write out of input matrix.", 2 );
1209  qoffset_x = qtm( 0, 3 );
1210  qoffset_y = qtm( 1, 3 );
1211  qoffset_z = qtm( 2, 3 );
1212  COMMENT( "load 3x3 matrix into local variables.", 2 );
1213  double r11 = qtm( 0, 0 );
1214  double r12 = qtm( 0, 1 );
1215  double r13 = qtm( 0, 2 );
1216  double r21 = qtm( 1, 0 );
1217  double r22 = qtm( 1, 1 );
1218  double r23 = qtm( 1, 2 );
1219  double r31 = qtm( 2, 0 );
1220  double r32 = qtm( 2, 1 );
1221  double r33 = qtm( 2, 2 );
1222  COMMENT( "compute lengths of each column; these determine grid spacings.", 2 );
1223  double xd = sqrt( r11 * r11 + r21 * r21 + r31 * r31 );
1224  double yd = sqrt( r12 * r12 + r22 * r22 + r32 * r32 );
1225  double zd = sqrt( r13 * r13 + r23 * r23 + r33 * r33 );
1226  COMMENT( "if a column length is zero, patch the trouble.", 2 );
1227  if( xd == 0.0 ) {
1228  r11 = 1.0;
1229  r21 = r31 = 0.0;
1230  xd = 1.0;
1231  }
1232  if( yd == 0.0 ) {
1233  r22 = 1.0;
1234  r12 = r32 = 0.0;
1235  yd = 1.0;
1236  }
1237  if( zd == 0.0 ) {
1238  r33 = 1.0;
1239  r13 = r23 = 0.0;
1240  zd = 1.0;
1241  }
1242  COMMENT( "These are relative to the native space. It is not pixdim, as pixdim refers to the image space.", 2 );
1243  /*
1244  * pixdim[ 0 ] = xd;
1245  * pixdim[ 1 ] = yd;
1246  * pixdim[ 2 ] = zd;
1247  */
1248  COMMENT( "normalize the columns.", 2 );
1249  r11 /= xd; r21 /= xd; r31 /= xd;
1250  r12 /= yd; r22 /= yd; r32 /= yd;
1251  r13 /= zd; r23 /= zd; r33 /= zd;
1252  /* At this point, the matrix has normal columns, but we have to allow
1253  * for the fact that the hideous user may not have given us a matrix
1254  * with orthogonal columns.
1255  * So, now find the orthogonal matrix closest to the current matrix.
1256  * One reason for using the polar decomposition to get this
1257  * orthogonal matrix, rather than just directly orthogonalizing
1258  * the columns, is so that inputting the inverse matrix to R
1259  * will result in the inverse orthogonal matrix at this point.
1260  * If we just orthogonalized the columns, this wouldn't necessarily hold.
1261  */
1262  Q( 0, 0 ) = r11; Q( 0, 1 ) = r12; Q( 0, 2 ) = r13;
1263  Q( 1, 0 ) = r21; Q( 1, 1 ) = r22; Q( 1, 2 ) = r23;
1264  Q( 2, 0 ) = r31; Q( 2, 1 ) = r32; Q( 2, 2 ) = r33;
1265 
1266  /* P is orthog matrix closest to Q */
1267  Matrix< double > P( MatrixOp::Polar( const_cast< const Matrix< double > & >( Q ) ) );
1268  r11 = P( 0, 0 ); r12 = P( 0, 1 ); r13 = P( 0, 2 ); /* unload */
1269  r21 = P( 1, 0 ); r22 = P( 1, 1 ); r23 = P( 1, 2 );
1270  r31 = P( 2, 0 ); r32 = P( 2, 1 ); r33 = P( 2, 2 );
1271  /* @param [ r11 r12 r13 ]
1272  * at this point, the matrix [ r21 r22 r23 ] is orthogonal
1273  * @param [ r31 r32 r33 ]
1274  */

1276  zd = MatrixOp::Determinant( P ); /* should be -1 or 1 */
1277  if( zd > 0 ) { /* proper */
1278  qfac = 1.0;
1279  }
1280  else { /* improper ==> flip 3rd column */
1281  qfac = -1.0;
1282  r13 = -r13; r23 = -r23; r33 = -r33;
1283  }
1284  COMMENT( "Compute quaternion parameters.", 2 );
1285  double a = r11 + r22 + r33 + 1.0;
1286  double b, c, d;
1287  if( a > 0.5 ) { /* simplest case */
1288  a = 0.5 * sqrt( a );
1289  b = 0.25 * ( r32 - r23 ) / a;
1290  c = 0.25 * ( r13 - r31 ) / a;
1291  d = 0.25 * ( r21 - r12 ) / a;
1292  }
1293  else { /* trickier case */
1294  xd = 1.0 + r11 - ( r22 + r33 ); /* 4*b*b */
1295  yd = 1.0 + r22 - ( r11 + r33 ); /* 4*c*c */
1296  zd = 1.0 + r33 - ( r11 + r22 ); /* 4*d*d */
1297  if( xd > 1.0 ) {
1298  b = 0.5 * sqrt( xd );
1299  c = 0.25 * ( r12 + r21 ) / b;
1300  d = 0.25 * ( r13 + r31 ) / b;
1301  a = 0.25 * ( r32 - r23 ) / b;
1302  }
1303  else if( yd > 1.0 ) {
1304  c = 0.5 * sqrt( yd );
1305  b = 0.25 * ( r12 + r21 ) / c;
1306  d = 0.25 * ( r23 + r32 ) / c;
1307  a = 0.25 * ( r13 - r31 ) / c;
1308  }
1309  else {
1310  d = 0.5 * sqrt( zd );
1311  b = 0.25 * ( r13 + r31 ) / d;
1312  c = 0.25 * ( r23 + r32 ) / d;
1313  a = 0.25 * ( r21 - r12 ) / d;
1314  }
1315  if( a < 0.0 ) {
1316  b = -b;
1317  c = -c;
1318  d = -d;
1319  a = -a;
1320  }
1321  }
1322  quatern_b = b;
1323  quatern_c = c;
1324  quatern_d = d;
