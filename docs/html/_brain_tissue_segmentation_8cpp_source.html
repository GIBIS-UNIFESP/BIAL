44  }

46  std::string msg( BIAL_ERROR( "Image and label dimensions do not match." ) );
47  throw( std::logic_error( msg ) );

49  size_t nlabels = static_cast< size_t >( Intensity::Maximum( Image< D >( label ), mask ) ) + 1;
50  Vector< double > mean( nlabels, 0.0 );
51  Vector< size_t > size( nlabels, 0 );
52  COMMENT( "Computing mean brightness and size of every label region.", 0 );
53  for( size_t pxl = 0; pxl < mask.size( ); ++pxl ) {
54  if( mask[ pxl ] != 0 ) {
55  mean( label[ pxl ] ) += image[ pxl ];
56  ++size( label[ pxl ] );
57  }
58  }
59  for( size_t lbl = 0; lbl < nlabels; ++lbl ) {
60  if( size( lbl ) != 0 ) {
61  mean( lbl ) /= size( lbl );
62  }
63  }
64  COMMENT( "Sorting mean vector.", 0 );
65  Vector< size_t > order = Sorting::Sort( mean );
66  COMMENT( "Computing position.", 0 );
67  Vector< size_t > position( order );
68  for( size_t idx = 0; idx < mean.size( ); ++idx ) {
69  position( order( idx ) ) = idx;
70  }
71  COMMENT( "Locating first high intensity cluster.", 0 );
72  size_t low_limit = 1;
73  COMMENT( "Finding first non-zero size cluster.", 0 );
74  while( ( low_limit < nlabels - 1 ) && ( mean( low_limit ) < intensity ) ) {
75  ++low_limit;
76  }
77  COMMENT( "Labeling samples into low and high intensity classes.", 0 );
78  for( size_t pxl = 0; pxl < image.size( ); ++pxl ) {
79  if( mask[ pxl ] != 0 ) {
80  if( position( label[ pxl ] ) < low_limit ) {
81  label[ pxl ] = 0;
82  }
83  else {
84  label[ pxl ] = 1;
85  }
86  }
