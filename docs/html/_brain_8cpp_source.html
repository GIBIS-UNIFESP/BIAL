47  void Brain::IntensityTissueLabeling( const Image< D > &image, const Image< D > &mask, Image< int > &label,
48  D intensity ) {
49  try {
50  if( image.Dims( ) != mask.Dims( ) ) {
51  std::string msg( BIAL_ERROR( "Image and mask dimensions do not match." ) );
52  throw( std::logic_error( msg ) );
53  }
54  if( image.Dims( ) != label.Dims( ) ) {
55  std::string msg( BIAL_ERROR( "Image and label dimensions do not match." ) );
56  throw( std::logic_error( msg ) );
57  }
58  size_t nlabels = static_cast< size_t >( Intensity::Maximum( Image< D >( label ), mask ) ) + 1;
59  Vector< double > mean( nlabels, 0.0 );
60  Vector< size_t > size( nlabels, 0 );
61 
62  COMMENT( "Computing mean brightness and size of every label region.", 0 );
63  for( size_t pxl = 0; pxl < mask.size( ); ++pxl ) {
64  if( mask[ pxl ] != 0 ) {
65  mean( label[ pxl ] ) += image[ pxl ];
66  ++size( label[ pxl ] );
67  }
68  }
69  for( size_t lbl = 0; lbl < nlabels; ++lbl ) {
70  if( size( lbl ) != 0 ) {
71  mean( lbl ) /= size( lbl );
72  }
73  }
74  COMMENT( "Sorting mean vector.", 0 );
75  Vector< size_t > order = Sorting::Sort( mean );
76 
77  COMMENT( "Computing position.", 0 );
78  Vector< size_t > position( order );
79  for( size_t idx = 0; idx < mean.size( ); ++idx ) {
80  position( order( idx ) ) = idx;
81  }
82  COMMENT( "Locating first high intensity cluster.", 0 );
83  size_t low_limit = 1;
84  COMMENT( "Finding first non-zero size cluster.", 0 );
85  while( ( low_limit < nlabels - 1 ) && ( mean( low_limit ) < intensity ) ) {
86  ++low_limit;
87  }
88  COMMENT( "Labeling samples into low and high intensity classes.", 0 );
89  for( size_t pxl = 0; pxl < image.size( ); ++pxl ) {
90  if( mask[ pxl ] != 0 ) {
91  if( position( label[ pxl ] ) < low_limit ) {
92  label[ pxl ] = 0;
93  }
94  else {
95  label[ pxl ] = 1;
96  }
97  }
98  }
99  }
100  catch( std::bad_alloc &e ) {
101  std::string msg( e.what( ) + std::string( "\n" ) + BIAL_ERROR( "Memory allocation error." ) );
102  throw( std::runtime_error( msg ) );

245  Image< D > Brain::RemoveOutliers( const Image< D > &img, const Image< D > &msk, float compression,
246  MRIModality modality ) {

252 
253  COMMENT( "Creating mask free from outiliers.", 0 );
254  if( modality == MRIModality::T1 ) {
255  COMMENT( "Detecting outliers threshold by hysteresis.", 0 );
256  size_t kernel_size = 5 + std::round( hst.size( ) / 1000.0 );
257  D hgh_intensity = SignalOp::HighHysteresis( hst, kernel_size, 0.1, 0.2 );
258  COMMENT( "hgh_intensity: " << hgh_intensity << ", max: " << Maximum( img * msk ), 2 );
259  for( size_t pxl = 0; pxl < img.size( ); ++pxl ) {
260  if( img[ pxl ] > hgh_intensity ) {
261  res[ pxl ] = 0;
262  }
263  }
264  }
265  else {
266  COMMENT( "Detecting outliers threshold by hysteresis.", 0 );
267  size_t kernel_size = 5 + std::round( hst.size( ) / 1000.0 );
268  D low_intensity = SignalOp::LowHysteresis( hst, kernel_size, 0.1, 0.2 );
269  for( size_t pxl = 0; pxl < img.size( ); ++pxl ) {
270  if( img[ pxl ] < low_intensity ) {
271  res[ pxl ] = 0;
272  }
273  }
274  }
275  COMMENT( "Eroding to avoid outliers neighbors.", 0 );
276  res = Morphology::ErodeBin( res, AdjacencyType::HyperSpheric( 5.0 / compression, img.Dims( ) ) );
277  return( res );
278  }
