151  bool Triangle::Intersect( const Ray &ray, float *tHit, float *rayEpsilon, DifferentialGeometry *dg ) const {
152  /* Compute s1. */
153  const Point3D &p1 = mesh->p[ v[ 0 ] ];
154  const Point3D &p2 = mesh->p[ v[ 1 ] ];
155  const Point3D &p3 = mesh->p[ v[ 2 ] ];
156  Vector3D e1 = p2 - p1;
157  Vector3D e2 = p3 - p1;
158  Vector3D s1 = Cross( ray.d, e2 );
159  float divisor = Dot( s1, e1 );
160  if( divisor == 0 ) {
161  return( false );
162  }
163  float invDivisor = 1.f / divisor;
164  /* Compute first baricentric coordinate. */
165  Vector3D d = ray.o - p1;
166  float b1 = Dot( d, s1 ) * invDivisor;
167  if( ( b1 < 0. ) || ( b1 > 1. ) ) {
168  return( false );
169  }
170  /* Compute second baricentric coordinate. */
171  Vector3D s2 = Cross( d, e1 );
172  float b2 = Dot( ray.d, s2 ) * invDivisor;
173  if( ( b2 < 0. ) || ( b2 > 1. ) ) {
174  return( false );
175  }
176  /* Compute t to intersection point. */
177  float t = Dot( e2, s2 ) * invDivisor;
178  if( ( t < ray.mint ) || ( t > ray.maxt ) ) {
179  return( false );
180  }
181  /* Compute triangle partial derivates. */
182  Vector3D dpdu, dpdv;
183  float uvs[ 3 ][ 2 ];
184  GetUVs( uvs );
185  /* Compute deltas for triangle partial derivatives */
186  float du1 = uvs[ 0 ][ 0 ] - uvs[ 2 ][ 0 ];
187  float du2 = uvs[ 1 ][ 0 ] - uvs[ 2 ][ 0 ];
188  float dv1 = uvs[ 0 ][ 1 ] - uvs[ 2 ][ 1 ];
189  float dv2 = uvs[ 1 ][ 1 ] - uvs[ 2 ][ 1 ];
190  Vector3D dp1 = p1 - p3;
191  Vector3D dp2 = p2 - p3;
192  float determinant = du1 * dv2 - dv2 * du2;
193  if( determinant == 0.f ) {
194  /* Handle zero determinant for triangle partial derivative matrix. */
195  CoordinateSystem( Cross( e2, e1 ).Normalized( ), &dpdu, &dpdv );
196  }
197  else {
198  float invdet = 1.f / determinant;
199  dpdu = ( dv2 * dp1 - dv1 * dp2 ) * invdet;
200  dpdv = ( -du2 * dp1 + du1 * dp2 ) * invdet;
201  }
202  /* Interpolate (u,v) triangle parametric coordinates. */
203  float b0 = 1 - b1 - b2;
204  float tu = b0 * uvs[ 0 ][ 0 ] + b1 * uvs[ 1 ][ 0 ] + b2 * uvs[ 2 ][ 0 ];
205  float tv = b0 * uvs[ 0 ][ 1 ] + b1 * uvs[ 1 ][ 1 ] + b2 * uvs[ 2 ][ 1 ];
206  /* Test intersection against alpha texture, if present. */
207 /*
208  * if(mesh->alphaTexture) {
209  * DifferentialGeometry dgLocal(ray(t),dpdu,dpdv, Normal(0,0,0), Normal(0,0,0),
210  * tu, tv, this);
211  * if(mesh->alphaTexture->Evaluate(dgLocal) == 0.0f){
212  * return false;
213  * }
214  * }
215  */
216  /* Fill in DifferentialGeometry from triangle hit */
217  *dg = DifferentialGeometry( ray( t ), dpdu, dpdv, Normal( 0, 0, 0 ),
218  Normal( 0, 0, 0 ), tu, tv, this );
219  *tHit = t;
220  *rayEpsilon = 1e-3f * ( *tHit );
221  return( true );
222  }
