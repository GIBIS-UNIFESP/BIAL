65 int PictureHandler::GetEncoderClsid(const WCHAR* format, CLSID* pClsid)
66 {
67  UINT num = 0; // number of image encoders
68  UINT size = 0; // size of the image encoder array in bytes
69 
70  ImageCodecInfo* pImageCodecInfo = NULL;
71 
72  GetImageEncodersSize(&num, &size);
73  if(size == 0)
74  return -1; // Failure
75 
76  pImageCodecInfo = (ImageCodecInfo*)(malloc(size));
77  if(pImageCodecInfo == NULL)
78  return -1; // Failure
79 
80  GetImageEncoders(num, size, pImageCodecInfo);
81 
82  for(UINT j = 0; j < num; ++j)
83  {
84  if( wcscmp(pImageCodecInfo[j].MimeType, format) == 0 )
85  {
86  *pClsid = pImageCodecInfo[j].Clsid;
87  free(pImageCodecInfo);
88  return j; // Success
89  }
90  }
91 
92  free(pImageCodecInfo);
93  return -1; // Failure
94 }
