135  double Graph< GRAPH_ADJACENCY >::PDF( size_t scl ) {
136  try {
137  COMMENT( "Computing sigma, i.e. the variance of gaussian.", 2 );
138  double sigma = adjacency.Sigma( scl );
139  if( sigma == 0.0 ) {
140  sigma = 1.0;
141  }
142  double mindens = std::numeric_limits< double >::max( );
143  double maxdens = std::numeric_limits< double >::min( );
144 
145  COMMENT( "Sigma: " << sigma << ". Computing nodes density and the minimal and maximal densities.", 2 );
146  for( size_t src = 0; src < density.size( ); ++src ) {
147  COMMENT( "src: " << src, 4 );
148  density( src ) = 1.0;
149  COMMENT( "Computing the density from the adjacents.", 4 );
150  size_t neighbors = adjacency.Arcs( src, scl );
151  for( size_t adj = 0; adj < neighbors; ++adj ) {
152  double weight = adjacency.ArcWeight( src, adj );
153  density( src ) += exp( -weight * weight / sigma );
154  COMMENT( "adj: " << adj << " weight: " << weight << ", exp: " << -weight * weight / sigma << ", dens: " <<
155  exp( -weight * weight / sigma ), 4 );
156  COMMENT( "partial sum: " << density( src ), 4 );
157  }
158  COMMENT( "Here, see the difference when additional arcs from symmetry are inserted.", 4 );
159  /* density( src ) /= ( neighbors + 1 ); */
160  COMMENT( "final density: " << density( src ), 4 );
161  COMMENT( "Getting maximum and minimum density for delta computing.", 4 );
162  if( mindens > density( src ) ) {
163  mindens = density( src );
164  }
165  if( maxdens < density( src ) ) {
166  maxdens = density( src );
167  }
168  }
169  COMMENT( "Density: " << density, 3 );
170 
171  COMMENT( "Computing IFT bucket size.", 2 );
172  if( maxdens == mindens ) { /* Hardly happens. */
173  return( maxdens / 10000.0 );
174  }
175  else {
176  return( ( maxdens - mindens ) / 10000.0 );
177  }
178  }
179  catch( std::bad_alloc &e ) {

264  Vector< double > external_weight( nlabels ); /* acumulate weights between the class and a distinct one */
265 
266  COMMENT( "Computing internal and external weights. nlabels: " << nlabels << ", elements: " << elements, 2 );
267  for( size_t src = 0; src < elements; ++src ) {
268  size_t src_lbl = label( src );
269  size_t neighbors = adjacency.Arcs( src, scl );

271  for( size_t ngb = 0; ngb < neighbors; ++ngb ) {
272  size_t tgt_lbl = label( adjacency.Arc( src, ngb ) );
273  double weight = adjacency.ArcWeight( src, ngb );
274  if( src_lbl == tgt_lbl ) {
275  internal_weight( src_lbl ) += ( 1.0 / ( 1.0 + weight ) ); /* intra-class weight */
276  }
