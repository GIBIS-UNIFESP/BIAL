53  double FuzzyCMeans< D >::MembershipUpdate( ) {
54  try {
55  Vector< double > distances( feats.Elements( ) * centroids.Elements( ), 0.0 );
56  COMMENT( "Computing mean distances.", 0 );
57  for( size_t ctd_a = 0; ctd_a < centroids.Elements( ); ++ctd_a ) {
58  for( size_t elm = 0; elm < feats.Elements( ); ++elm ) {
59  for( size_t ftr = 0; ftr < feats.Features( ); ++ftr ) {
60  distances( elm * centroids.Elements( ) + ctd_a ) +=
61  static_cast< double >( ( feats( elm, ftr ) - centroids( ctd_a, ftr ) ) *
62  ( feats( elm, ftr ) - centroids( ctd_a, ftr ) ) );
63  }
64  distances( elm * centroids.Elements( ) + ctd_a ) = sqrt( distances( elm * centroids.Elements( ) + ctd_a ) );
65  }
66  }
67  double max = 0.0;
68  COMMENT( "Updating memberships.", 0 );
69  for( size_t ctd_a = 0; ctd_a < centroids.Elements( ); ++ctd_a ) {
70  double val = 0.0;
71  for( size_t elm = 0; elm < feats.Elements( ); ++elm ) {
72  if( distances( elm * centroids.Elements( ) + ctd_a ) <= 0.0000001 ) {
73  val = 1.0;
74  }
75  else {
76  for( size_t ctd_b = 0; ctd_b < centroids.Elements( ); ++ctd_b ) {
77  if( distances( elm * centroids.Elements( ) + ctd_b ) <= 0.0000001 ) {
78  val = 9.0e99;
79  break;
80  }
81  val +=
82  pow( distances( elm * centroids.Elements( ) + ctd_a ) /
83  distances( elm * centroids.Elements( ) + ctd_b ), 2.0 / ( m - 1.0 ) );
84  }
85  val = 1.0 / val;
86  }
87  if( max < membership( ctd_a, elm ) - val ) {
88  max = membership( ctd_a, elm ) - val;
89  }
90  membership( ctd_a, elm ) = val;
91  }
92  }
93  return( max );
94  }
95  catch( std::bad_alloc &e ) {
96  std::string msg( e.what( ) + std::string( "\n" ) + BIAL_ERROR( "Memory allocation error." ) );

114  void FuzzyCMeans< D >::CentroidUpdate( ) {
115  try {
116  Vector< double > sum( feats.Features( ) );
117  for( size_t ctd = 0; ctd < centroids.Elements( ); ++ctd ) {
118  double sum2 = 0.0;
119  for( size_t ftr = 0; ftr < feats.Features( ); ++ftr ) {
120  sum( ftr ) = 0.0;
121  for( size_t elm = 0; elm < feats.Elements( ); ++elm ) {
122  double power = pow( static_cast< double >( membership( ctd, elm ) ), m );
123  sum( ftr ) += power * static_cast< double >( feats( elm, ftr ) );
124  if( ftr == 0 ) {
125  sum2 += power;
126  }
127  }
128  if( ( sum2 > -0.0000001 ) && ( sum2 < 0.0000001 ) ) {
129  std::string msg( BIAL_ERROR( "Division by zero while computing centroids." ) );
130  throw( std::logic_error( msg ) );
131  }
132  centroids( ctd, ftr ) = static_cast< D >( sum( ftr ) / sum2 );

201  centroids( 0, ftr ) = feats( 0, ftr );
202  centroids( centroids.Elements( ) - 1, ftr ) = feats( 0, ftr );
203  for( size_t elm = 0; elm < feats.Elements( ); ++elm ) {
204  if( centroids( 0, ftr ) > feats( elm, ftr ) ) {
205  centroids( 0, ftr ) = feats( elm, ftr );
206  }
207  if( centroids( centroids.Elements( ) - 1, ftr ) < feats( elm, ftr ) ) {
208  centroids( centroids.Elements( ) - 1, ftr ) = feats( elm, ftr );
