237  Image< D > sub_msk = ImageOp::Resize( msk, low_coord, hgh_coord );
238  sub_img = Geometrics::Scale( sub_img, 1.0 / compression, true );
239  sub_msk = Geometrics::Scale( sub_msk, 1.0 / compression, true );
240  sub_img = ImageOp::AddFrame( sub_img, 2 * compression );
241  sub_msk = ImageOp::AddFrame( sub_msk, 2 * compression );
242  COMMENT( "Computing mask without outliers.", 0 );

244  sub_msk = Morphology::DilateBin( sub_msk, AdjacencyType::HyperSpheric( 1.9, img.Dims( ) ) );
245  COMMENT( "Computing bias surface.", 0 );
246  Image< D > bias = BiasSurfaceEstimation( sub_img, sub_msk, cln_msk, radius / compression, modality );
247  COMMENT( "Converting bias image to the input format.", 0 );
248  Vector< size_t > unframed_size( hgh_coord );
249  for( size_t dms = 0; dms < unframed_size.size( ); ++dms )
250  unframed_size( dms ) -= low_coord( dms ) - 1;
