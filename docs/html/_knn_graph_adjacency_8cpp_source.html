195  try {

197  size_t size = std::min( feature.Elements( ), MAX_SAMPLES ) - 1;
198  COMMENT( "Computing the power that satisfies: ( 0.1*size^( 1/p ) )^p = 2, so r = 0.1 implies k = 2.", 1 );
199  float power = std::log( 2.0f / size ) / std::log( 0.1f );
200  COMMENT( "size: " << size << ", power: " << power << ". Getting the required element indexes.", 1 );
201  size_t min_index = static_cast< size_t >( std::pow( std::pow( size, 1.0f / power ) * scl_min, power ) + 0.5f );
202  if( min_index == 0 ) {
203  min_index = 1;
204  }
205  size_t max_index = static_cast< size_t >( std::pow( std::pow( size, 1.0f / power ) * scl_max, power ) + 0.5f );
206  COMMENT( "min_index: " << min_index << ", max_index: " << min_index, 1 );
207  COMMENT( "Inserting elements into the resultant vector.", 1 );
208  size_t displacement = 1;

245  Vector< Vector< size_t > > &KnnGraphAdjacency::HeterogeneousAdjacency( const Vector< double > &density,
246  size_t scl, double delta ) {
247  try {
248  COMMENT( "Initializing heterogeneous adjacency.", 2 );
249  size_t elements = arc.size( 0 );
250  plateau = Vector< Vector< size_t > >( elements, Vector< size_t >( ) );
251  size_t neighbors = scale( scl );
252  COMMENT( "Add arcs to guarantee symmetry on plateaus.", 2 );
253  for( size_t src = 0; src < elements; ++src ) {
254  for( size_t knn = 0; knn < neighbors; ++knn ) {
255  size_t tgt = arc( src, knn );
256  if( ( density( src ) + delta >= density( tgt ) ) &&
257  ( density( src ) - delta <= density( tgt ) ) ) {
258  COMMENT( "Insert src in heterogeneous adjacency of tgt, if src is not in knn adjacency of tgt.", 3 );
259  bool insert_src = true;
260  for( size_t bck = 0; bck < neighbors; ++bck ) {
261  if( arc( tgt, bck ) == src ) {
262  insert_src = false;
263  break;
264  }
265  }
266  if( insert_src ) {
267  plateau( tgt ).push_back( src );
268  }
269  }
270  }
271  }
